import pandas as pd

def extract_and_clean_epl_data(file_path):
    """
    Loads raw EPL match data, cleans missing values, and formats team names.
    """
    print("Loading raw EPL dataset...")
    try:
        # Load the raw data
        df = pd.read_csv(file_path)
        
        # Clean team names (example: standardizing Manchester United)
        df['HomeTeam'] = df['HomeTeam'].replace({'Man United': 'Manchester United', 'MANUTD': 'Manchester United'})
        df['AwayTeam'] = df['AwayTeam'].replace({'Man United': 'Manchester United', 'MANUTD': 'Manchester United'})
        
        # Handle missing values in critical columns
        df.dropna(subset=['HomeTeam', 'AwayTeam', 'FTHG', 'FTAG'], inplace=True)
        
        print(f"Data cleaned successfully. Total matches processed: {len(df)}")
        return df

    except FileNotFoundError:
        print("Error: The specified data file was not found. Please check the path.")
        return None
